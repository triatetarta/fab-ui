## Usage

```tsx
import { Fieldset, FieldsetLegend } from "@/components/ui/fieldset"
```

```tsx
<Fieldset>
  <FieldsetLegend>Personal Information</FieldsetLegend>
  <Field>
    <FieldLabel>Name</FieldLabel>
    <Input placeholder='John Doe' />
  </Field>
</Fieldset>
```
