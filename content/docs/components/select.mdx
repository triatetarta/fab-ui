## Usage

```tsx
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select"
```

```tsx
<Select>
  <SelectTrigger>
    <SelectValue placeholder='Select an option' />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value='option-1'>Option 1</SelectItem>
    <SelectItem value='option-2'>Option 2</SelectItem>
  </SelectContent>
</Select>
```
